ImagePreview 图片预览

单图放大

<div class="javaex-image" style="width:200px;height:200px;">
	<a href="javascript:;" onclick="img(this)">
		<img class="javaex-img" fit="cover" src="https://webimg.javaex.cn/Ft0-o8l5oCgH9dQUV67eMVODZVU_" />
	</a>
</div>

<script>
	// obj为当前点击的元素对象
	function img(obj) {
		javaex.imagePreview({
			type : "image",	// 弹出层类型
			url : obj.children[0].src	// 图片地址
		});
	}
	
	// 或者下面这种写法
	$("img").click(function() {
		javaex.imagePreview({
			type : "image",	// 弹出层类型
			url : $(this).attr("src")	// 图片地址
		});
	});
</script>

参数说明

参数 默认值 说明 版本
type 必须是image
url 图片地址，就按照示例中的写
opend 打开弹窗后的回调函数
opend : function() {
	console.log("弹窗已打开");
}

<script>
	// 请按实际需求填写jquery选择器，最终指定到img即可
	$("#demo li img").click(function() {
		javaex.imagePreview({
			type : "images",	// 弹出层类型
			selector : "#demo li img",	// jquery选择器，和上面保持一致
			url : $(this).attr("src")	// 图片地址，就这么写
		});
	});
</script>

参数说明

参数 默认值 说明 版本
type 必须是images
selector jquery选择器，和调用的选择器保持一致
url 图片地址，就按照示例中的写
placeholder data:image/png;base64... 图片延迟加载时的占位内容
dataOriginal data-original 图片延迟加载时，从该属性中获取真正的图片地址
opend 打开弹窗后的回调函数
opend : function() {
	console.log("弹窗已打开");
}
