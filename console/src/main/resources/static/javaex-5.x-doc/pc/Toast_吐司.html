Toast 吐司

基础用法

<button id="demo" class="javaex-btn primary">点击测试</button>

<script>
	$("#demo").click(function() {
		javaex.toast({
			content : "我是来测试的"
		});
	});
</script>

参数说明

参数 默认值 说明 版本
content 弹出层内容
live 2000 提示存在时间，单位:毫秒
opend 打开后的回调函数
opend : function() {
	console.log("弹窗已打开");
}

定位用法

<button class="javaex-btn success" onclick="toast1(this)">成功</button>
<button class="javaex-btn danger" onclick="toast2(this)">失败</button>

<script>
	// obj为当前点击的元素对象，obj是必须的
	function toast1(obj) {
		javaex.toast(obj, {
			content : "关注成功",
			type : "success"
		});
	}

	function toast2(obj) {
		javaex.toast(obj, {
			content : "不能关注自己",
			type : "error"
		});
	}
</script>

参数说明

参数 默认值 说明 版本
content 弹出层内容
type success success:成功。 error:失败
live 2000 提示存在时间，单位:毫秒
offsetTop 0 上偏移量，单位：像素
opend 打开后的回调函数
opend : function() {
	console.log("弹窗已打开");
}
