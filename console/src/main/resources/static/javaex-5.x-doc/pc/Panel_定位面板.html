Panel 定位面板

自定义内容弹出层（可以使用 javaex.close(); 关闭）

<button class="javaex-btn primary" onclick="panel1(this)" style="margin-left:200px;">上</button>
<button class="javaex-btn primary" onclick="panel2(this)" style="margin-left:200px;">下-左对齐</button>
<button class="javaex-btn primary" onclick="panel3(this)" style="margin-left:200px;">下-右对齐</button>
<button class="javaex-btn primary" onclick="panel4(this)" style="margin-left:200px;">左</button>
<button class="javaex-btn primary" onclick="panel5(this)" style="margin-left:200px;">右</button>

<script>
	// 内容随意，这边只是演示而已
	var html = '';
	html += '<div class="javaex-panel-box">';
	html += '	<ul>';
	html += '		<p>排序</p>';
	html += '		<li>创建时间</li>';
	html += '		<li>修改时间</li>';
	html += '		<li>笔记名称</li>';
	html += '		<hr class="javaex-divider"></hr>';
	html += '		<p>查看方式</p>';
	html += '		<li>摘要</li>';
	html += '		<li>列表</li>';
	html += '	</ul>';
	html += '</div>';

	// obj为当前点击的元素对象
	function panel1(obj) {
		// obj是必须的
		javaex.panel(obj, {
			position : "up",
			content : html
		});
	}
	
	function panel2(obj) {
		// obj是必须的
		javaex.panel(obj, {
			position : "down",
			content : html
		});
	}
	
	function panel3(obj) {
		// obj是必须的
		javaex.panel(obj, {
			position : "down",
			alignment : "right",
			content : html
		});
	}
	
	function panel4(obj) {
		// obj是必须的
		javaex.panel(obj, {
			position : "left",
			content : html
		});
	}
	
	function panel5(obj) {
		// obj是必须的
		javaex.panel(obj, {
			position : "right",
			content : html
		});
	}
</script>

参数说明

参数 默认值 说明 版本
position right 弹出层位置（up：上；down：下；left：左；right：右）
alignment left 对齐方式，仅在 position=="down" 时有效
content 弹出层内容，html代码。不局限于示例中的菜单写法
offsetLeft 0 左偏移量
offsetTop 0 上偏移量
callback 打开弹窗后的回调函数
callback : function() {
	console.log("弹窗已打开");
}
