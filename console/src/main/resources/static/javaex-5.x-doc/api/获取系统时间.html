获取系统时间

方法 说明 版本 
var now = javaex.now();
 返回当前时间，格式yyyy-MM-dd HH:mm:ss 
var year = javaex.getYear();
 返回当前年份 
var month = javaex.getMonth();
 返回当前月份 
var day = javaex.getDay();
 返回当前日 
var today = javaex.getToday();
 返回今日日期，格式yyyy-MM-dd 
var week = javaex.getWeek();
 返回当日所属星期（1=周一，2=周二...7=周日） 
/* 
 * 获得时间差
 * startTime : 开始时间（时间格式为 yyyy-MM-dd HH:mm:ss 例如：2018-06-21 00:00:00）
 * endTime : 结束时间（时间格式为 yyyy-MM-dd HH:mm:ss 例如：2018-06-21 00:00:00）
 * type : 返回精度（second，minute，hour，day）
 */
var diff = javaex.getTimeDiff(startTime, endTime, type);
 获得时间差 
/* 
 * 获得两个日期之间的周次数
 * startDate : 开始日期（时间格式为 yyyy-MM-dd 例如：2021-09-01）
 * endDate : 结束日期（时间格式为 yyyy-MM-dd 例如：2022-06-28）
 */
var diff = javaex.getWeekDiff(startDate, endDate);
 获得两个日期之间的周次数
