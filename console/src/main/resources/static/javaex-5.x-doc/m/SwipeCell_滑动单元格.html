SwipeCell 滑动单元格（只能在移动端使用）

基础用法

<div class="javaex-swipe">
	<div class="javaex-swipe-content">
		<div class="javaex-cell">
			<div class="javaex-cell-value">左滑删除1</div>
		</div>
	</div>
	<div class="javaex-swipe-right">
		<div class="javaex-btn danger" square onclick="doDel()">
			<div class="javaex-button-warp">删除</div>
		</div>
	</div>
</div>

<div class="javaex-swipe">
	<div class="javaex-swipe-content">
		<div class="javaex-cell">
			<div class="javaex-cell-value">左滑删除2</div>
		</div>
	</div>
	<div class="javaex-swipe-right">
		<div class="javaex-btn danger" square onclick="doDel()">
			<div class="javaex-button-warp">删除</div>
		</div>
	</div>
</div>

<script>
	javaex.swipe({
		el : ".javaex-swipe"
	});

	function doDel() {
		javaex.toast({
			content : "点击了删除",
			type : "message"
		});
	}
</script>
