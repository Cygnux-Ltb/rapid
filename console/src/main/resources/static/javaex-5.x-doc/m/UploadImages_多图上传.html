UploadImages 多图上传

代码（base64示例）

<div class="javaex-uploader">
	<div class="javaex-uploader-wrapper">
		<div class="javaex-uploader-upload">
			<i class="javaex-badge-wrapper javaex-icon icon-camera"></i>
			<input type="file" id="upload" class="javaex-uploader-input" accept="image/*">
		</div>
	</div>
</div>

<script>
	javaex.uploadImages({
		id : "upload",			// <input type="file" />的id
		dataType : "base64",		// 返回的数据类型：base64 或 url
		imgList : ['http://webimg.javaex.cn/Fv7_P5dmShg1-deGM21gF8qCfSDL'],	// 图片回显
		maxNum : 3,	// 最多上传多少张图片
		callback : function (rtn) {
			console.log(rtn);
		},
		deleteImage : function() {
			console.log("做点什么吧");
		},
		chooseBefore : function() {
			console.log("选择文件前的回调函数");
		},
		chooseAfter : function(rtn) {
			console.log("选择文件后的回调函数");
			// 选择文件后的回调函数，会返回一些文件信息
			console.log(rtn.filename);
			console.log(rtn.filesize);
			console.log(rtn.fileSuffix);
		}
	});
</script>

代码（URL示例）

<div class="javaex-uploader">
	<div class="javaex-uploader-wrapper">
		<div class="javaex-uploader-upload">
			<i class="javaex-badge-wrapper javaex-icon icon-camera"></i>
			<input type="file" id="upload" class="javaex-uploader-input" accept="image/*">
		</div>
	</div>
</div>

<script>
	javaex.uploadImages({
		id : "upload",			// <input type="file" />的id
		dataType : "url",		// 返回的数据类型：base64 或 url
		image : {
			url : "https://upload.ximalaya.com/dtres/headerThumb/upload",	// 请求路径
			prefix : "",	// 前缀，如果后台返回的图片地址不是完整的链接，可以添加前缀
			imgUrl : "data.imgUrl"	// 根据返回的数据对象，获取图片地址。  例如后台返回的数据为：{code: "000000", message: "操作成功！", data: {imgUrl: "/1.jpg"}}
		},
		param : {
			"file" : "file",    // 参数名称，左边的 "file" 是固定必须的。   右边的 "file"是参数名，Spring中与MultipartFile的参数名保持一致
			"projectId" : "",    // 除了file这个key，其他都是MultipartFile以外的参数（自定义）
			"type" : "2"
		},
		imgList : ['http://webimg.javaex.cn/Fv7_P5dmShg1-deGM21gF8qCfSDL'],	// 图片回显
		maxNum : 3,	// 最多上传多少张图片
		callback : function (rtn) {
			console.log(rtn);
		},
		deleteImage : function() {
			console.log("做点什么吧");
		},
		chooseBefore : function() {
			console.log("选择文件前的回调函数");
		},
		chooseAfter : function(rtn) {
			console.log("选择文件后的回调函数");
			// 选择文件后的回调函数，会返回一些文件信息
			console.log(rtn.filename);
			console.log(rtn.filesize);
			console.log(rtn.fileSuffix);
		}
	});
</script>

参数说明

参数 默认值 说明 版本
id <input type="file" />的id
param {"file":"file"} 
param : {
	"file" : "file",    // 参数名称，左边的 "file" 是固定必须的。   右边的 "file"是参数名，Spring中与MultipartFile的参数名保持一致
	"projectId" : projectId,    // 除了file这个key，其他都是MultipartFile以外的参数（自定义）
	"type" : "2"
}
header 
header : {
	"token" : "123456",
	"token2" : "xxxx"
}
maxSize 单张图片的大小上限，单位KB，0或空为不限制
dataType base64 返回的数据类型：base64 或 url
填写url时，需要设置url请求地址
image 图片上传的一些属性设置
image : {
	url : "${pageContext.request.contextPath}/upload/upload.json",	// 请求路径
	prefix : "",		// 前缀，如果后台返回的图片地址不是完整的链接，可以添加前缀
	imgUrl : "data.imgUrl"	// 根据返回的数据对象，获取图片地址。  例如后台返回的数据为：{code: "000000", message: "操作成功！", data: {imgUrl: "/1.jpg"}}
}
imgList 回显图片（包含图片地址的 数组 或 List）
maxNum 10 最多上传多少张图片
sendCookie false 是否发送cookie
crossDomain true 是否允许跨域
callback dataType为base64时，返回图片的base64
dataType为url时，获取后台返回的数据
回显图片，或图片上传成功时会调用这个回调函数
deleteImage 删除图片的回调函数
chooseBefore 选择文件前的回调函数
chooseAfter 选择文件后的回调函数
