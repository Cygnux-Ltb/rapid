	</div>

关闭模式

<div class="javaex-noticebar">
	<div class="javaex-noticebar-left-icon">
		<i class="javaex-icon icon-volume-medium"></i>
	</div>
	<div class="javaex-noticebar-wrap">

	</div>

</div>

<script>
	function doDel(obj) {
		$(obj).parent().remove();
	}
</script>
