OptMenu 操作菜单

<button id="demo" class="javaex-btn default" block size="large">点击调用操作菜单</button>

<script>
	$("#demo").click(function() {
		let id = "abc";
		javaex.optMenu({
			title : "操作说明的标题",
			menus : {
				// 允许加图标等样式，但是最外层必须用单引号包裹
				'<span class="icon-trash"></span> 删除' : 'del(\''+ id +'\')',
				'<span class="icon-block"></span> 禁用' : 'disable(\''+ id +'\')'
			}
		});
	});

	function del(id) {
		alert("点击了删除，传参：" + id);
	}
	
	function disable(id) {
		alert("点击了禁用，传参：" + id);
	}
</script>

参数说明

参数 默认值 说明 版本
title 不为空时，会显示标题
menus 自定义菜单
